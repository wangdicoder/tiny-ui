```jsx live

  <Button btnType="primary">Button</Button>
  <Button>Another Button</Button>
  <PopConfirm title="Are you sure delete this task?" okText="Yes" cancelText="No">
    <Button>Confirm</Button>
  </PopConfirm>
</Space>
```
